% ─── slide 2: motivation ──────────────────────────────────────────────────────
\begin{frame}{What Is Rage-Bait, and Why Model It?}
  \begin{columns}[T]
    \begin{column}{0.33\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.80\textheight,keepaspectratio]{rage.png}
    \end{column}
    \begin{column}{0.63\textwidth}
      \small
      \textbf{Rage-bait} = content engineered to provoke outrage rather
      than genuine sharing.  It generates far more \emph{replies} (arguments)
      than retweets.

      \vspace{0.2em}
      Why does it matter?
      \begin{itemize}
        \item Emotionally charged content spreads faster on every major
              platform \cite{brady2017}.
        \item Platforms actively amplify it because it drives engagement
              metrics.
        \item But sustained toxicity drives users away and degrades
              platform health.
      \end{itemize}

      \vspace{0.15em}
      \begin{block}{Our question}
        Can we \emph{quantify} the tradeoff between amplification-driven
        engagement and its cost in toxicity and user loss?
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ─── slide 3: dataset ─────────────────────────────────────────────────────────
\begin{frame}{The Dataset: Higgs Boson Twitter Cascade}
  \begin{columns}[T]
    \begin{column}{0.58\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.72\textheight,keepaspectratio]{higgs_overview.png}
    \end{column}
    \begin{column}{0.38\textwidth}
      \small
      \textbf{What it is:}\\
      563\,069 retweets + 114\,000 replies around the Higgs boson
      announcement (4 to 7 July 2012).

      \vspace{0.5em}
      \textbf{Why it fits:}\\
      Science news, not politics, yet the reply surge relative to
      retweets mirrors the rage-bait pattern exactly.

      \vspace{0.5em}
      \begin{block}{Two signals we use}
        \begin{itemize}
          \item \textbf{Retweets} $\to$ calibrate spread\\
                ($\beta_0$ spread rate, $\gamma_0$ fatigue rate)
          \item \textbf{Replies} $\to$ proxy for\\
                discussion pressure $\tau$ (latent toxicity)
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ─── slide 4: model overview ──────────────────────────────────────────────────
\begin{frame}{The IVFS Model: Classifying \emph{Content}, Not Users}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.55\textheight,keepaspectratio]{ivfs_structure_diagram.png}
      \vspace{0.3em}
      {\scriptsize\itshape A deterministic compartmental ODE model: content is
      tracked through states, not individuals. Analogous to epidemic
      SIR models but applied to online content lifecycles.}
    \end{column}
    \begin{column}{0.50\textwidth}
      \small
      \begin{tabular}{@{}cl@{}}
        \toprule
        Variable & Meaning \\
        \midrule
        $I$ & \textbf{Ignored}: content not yet seen \\
        $V$ & \textbf{Viral}: actively spreading \\
        $F$ & \textbf{Fatigued}: stale, audience bored \\
        $S$ & \textbf{Suppressed}: platform removed \\[4pt]
        $\tau$ & Discussion pressure (toxicity proxy) \\
        $U$    & Active user pool \\
        \bottomrule
      \end{tabular}

      \vspace{0.6em}
      \textbf{Why content?}\\
      Content volumes are directly observable in data;
      individual user mental states are not.

      \vspace{0.3em}
      The amplification parameter $\alpha$ (platform amplif. strength) controls how strongly
      the platform boosts $I \!\to\! V$ transitions.
    \end{column}
  \end{columns}
\end{frame}

% ─── slide 6: equations ───────────────────────────────────────────────────────
\begin{frame}{The Full ODE System}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \small
      \begin{align*}
        \dot I &= \tfrac{\rho U}{1+U}
                 - \beta_{\mathrm{eff}}\,I\,V
                 - (\delta+\mu_c)\,I \\[2pt]
        \dot V &= \beta_{\mathrm{eff}}\,I\,V
                 - \gamma_{\mathrm{eff}}\,V
                 - \mu_c\,V \\[2pt]
        \dot F &= \gamma_{\mathrm{eff}}\,V - \mu_c\,F \\[2pt]
        \dot S &= \delta\,I - \mu_c\,S \\[4pt]
        \dot\tau &= \phi\,V - \psi\,\tau \\[2pt]
        \dot U &= \nu - \lambda_u(1+w\tau)\,U
      \end{align*}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{alertblock}{Toxicity coupling}
        \small
        \[
          \beta_{\mathrm{eff}} = \alpha\,\beta_0\,(1+\kappa\,\tau)
        \]
        \[
          \gamma_{\mathrm{eff}} = \gamma_0\,(1+\eta\,\tau)
        \]
        \vspace{-0.3em}
        $\kappa$ (spread coupling), $\eta$ (fatigue coupling).
      \end{alertblock}

      \vspace{0.15em}
      \small\textbf{Feedback} ($\phi$: pressure gen. rate):
      $V \xrightarrow{\phi} \tau \xrightarrow{\kappa} \beta_{\mathrm{eff}} \to V$

      \vspace{0.1em}
      Viral content $\to$ pressure $\to$ faster spread.

      \vspace{0.15em}
      \textbf{Brakes:} $\gamma_{\mathrm{eff}}$ (fatigue),
      $\psi$ (pressure decay), $w$ (tox. weight on departure).
    \end{column}
  \end{columns}
\end{frame}

% ─── slide 7: R0 and threshold ────────────────────────────────────────────────
\begin{frame}{The Reproduction Number $\mathcal{R}_0$ and Amplification Threshold}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{block}{Basic reproduction number}
        \[
          \mathcal{R}_0
          = \frac{\alpha\,\beta_0\,I_{\mathrm{DFE}}}{\gamma_0 + \mu_c}
        \]
        \scriptsize
        $\alpha$ amplification, $\beta_0$ spread rate,
        $I_{\mathrm{DFE}}$ ignored content at the DFE
        (disease-free equilibrium: the state where $V{=}0$, $\tau{=}0$),
        $\gamma_0$ fatigue rate, $\mu_c$ clearance rate.
        \vspace{0.1em}
        \small $\mathcal{R}_0$ = expected new viral pieces from one viral piece
        when virality is rare.
      \end{block}

      \vspace{0.2em}
      \small
      \begin{itemize}
        \item $\mathcal{R}_0 < 1$: every spike dies out on its own.
        \item $\mathcal{R}_0 > 1$: virality becomes \emph{endemic},
              persisting indefinitely.
      \end{itemize}

      \vspace{0.15em}
      \small\textbf{Well-posedness:} all state variables stay bounded
      and non-negative for all $t \geq 0$.
    \end{column}
    \begin{column}{0.50\textwidth}
      \begin{alertblock}{A sharp platform threshold}
        \[
          \boxed{\;\alpha_{\mathrm{crit}} \approx 0.3165\;}
        \]
        \small The value of $\alpha$ (amplification) where $\mathcal{R}_0 = 1$.
      \end{alertblock}

      \vspace{0.25em}
      \small
      \begin{itemize}
        \item \textbf{Below} $\alpha_{\mathrm{crit}}$: platform is safe,
              no sustained rage-bait, zero endemic toxicity, full user base.
        \item \textbf{Above} $\alpha_{\mathrm{crit}}$: a transcritical
              bifurcation occurs and a toxic endemic equilibrium appears.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% ─── slide 7: bifurcation ─────────────────────────────────────────────────────
\begin{frame}{Bifurcation Analysis: The Threshold Verified Numerically}
  \centering
  \includegraphics[width=0.90\linewidth,height=0.50\textheight,keepaspectratio]{r0_bifurcation.png}

  \vspace{0.2em}
  \begin{columns}[T]
    \begin{column}{0.38\textwidth}
      \small
      \textbf{Left:} $\mathcal{R}_0$ (reprod. number) linear in $\alpha$ (amplification).
      Crosses 1 at $\alpha_{\mathrm{crit}} \approx 0.3165$.
      \textbf{Right:} $V^*$ (steady-state viral volume) bifurcates from zero, classic transcritical shape.
    \end{column}
    \begin{column}{0.58\textwidth}
      \begin{block}{Stability}
        \scriptsize
        \begin{itemize}
          \item DFE (disease-free eq., $V{=}0$, $\tau{=}0$)
                eigenvalue for $V$: $(\gamma_0{+}\mu_c)(\mathcal{R}_0{-}1)$,
                negative iff $\mathcal{R}_0 < 1$
                ($\gamma_0$ fatigue rate, $\mu_c$ clearance rate).
          \item Endemic eq.: confirmed locally asymptotically stable
                for all tested $\alpha$ (amplif.) $> \alpha_{\mathrm{crit}}$.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ─── slide 11: conclusions ────────────────────────────────────────────────────
\begin{frame}{Conclusions}
  \begin{enumerate}
    \item \textbf{Sharp amplification threshold.}\;
          $\alpha_{\mathrm{crit}} \approx 0.3165$ separates a safe regime
          (no sustained virality) from a toxic endemic regime.

    \vspace{0.45em}
    \item \textbf{Self-reinforcing feedback loop.}\;
          Above threshold: viral content $\to$ discussion pressure
          $\to$ boosted spread $\to$ more viral content.

    \vspace{0.45em}
    \item \textbf{Quantifiable tradeoff.}\;
          Aggressive amplification (+73\,\% engagement) costs a 40\,\%
          smaller, more toxic, less sustainable user base.

    \vspace{0.45em}
    \item \textbf{IVFS outperforms SIR.}\;
          Spike $R^2 = 0.89$ vs.\ $0.69$; the content-lifecycle
          structure captures the long tail that SIR misses.
  \end{enumerate}
