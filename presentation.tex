\begin{frame}{Why Rage-Bait?}
  \begin{columns}[T]
    \begin{column}{0.34\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.85\textheight,keepaspectratio]{rage.png}
    \end{column}
    \begin{column}{0.62\textwidth}
      {\small Emotional, morally charged content spreads further and faster than
      neutral posts on every major platform \cite{brady2017}.
      Platforms know this, and they tune their algorithms to ride it.
      But what happens downstream?}

      \vspace{0.3em}
      {\small
      \textbf{Rage-bait} = content that gets people \emph{arguing},
      not just sharing. High replies relative to retweets
      $\;\Rightarrow\;$ high $\tau/V$.

      \vspace{0.2em}
      The Higgs boson discovery triggered exactly this:
      replies flooded in way beyond the retweet count.
      Science, not politics, same pattern.}

      \vspace{0.3em}
      \begin{block}{Central question}
        \small How does algorithmic amplification shape the full lifecycle of
        rage-bait, and what does it cost in toxicity and user retention?
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% slide 4a: content ODEs
\begin{frame}{Content Dynamics}
  \begin{align*}
    \dot I &= \frac{\rho\, U}{1+U}
             - \beta_{\mathrm{eff}}\,I\,V
             - (\delta + \mu_c)\,I \\[6pt]
    \dot V &= \beta_{\mathrm{eff}}\,I\,V
             - \gamma_{\mathrm{eff}}\,V
             - \mu_c\,V \\[6pt]
    \dot F &= \gamma_{\mathrm{eff}}\,V
             - \mu_c\,F \\[6pt]
    \dot S &= \delta\,I
             - \mu_c\,S
  \end{align*}

  \vspace{0.3em}
  \begin{alertblock}{Effective rates}
    $\beta_{\mathrm{eff}} = \alpha\,\beta_0\,(1+\kappa\,\tau)$
    \hfill
    $\gamma_{\mathrm{eff}} = \gamma_0\,(1+\eta\,\tau)$
  \end{alertblock}

  \vspace{0.3em}
  $\alpha$ = platform amplification strength. \quad
  $\kappa$ = toxicity boost to spread. \quad
  $\eta$ = toxicity-accelerated fatigue.
\end{frame}

% slide 4b: pressure and user ODEs
\begin{frame}{Pressure and User Dynamics}
  \begin{align*}
    \dot \tau &= \phi\,V - \psi\,\tau \\[8pt]
    \dot U    &= \nu - \lambda_u\,(1 + w\,\tau)\,U
  \end{align*}

  \vspace{0.4em}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{block}{Pressure equation}
        Viral content generates discussion pressure at rate $\phi$.

        Pressure decays naturally at rate $\psi$.
      \end{block}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{alertblock}{User equation}
        Users arrive at rate $\nu$ and leave at baseline rate $\lambda_u$.

        Toxicity multiplies departure by $(1 + w\,\tau)$.
      \end{alertblock}
    \end{column}
  \end{columns}

  \vspace{0.6em}
  \centering
  \textbf{Feedback:} $V \;\xrightarrow{\;\phi\;}\; \tau
  \;\xrightarrow{\;\kappa\;}\; \beta_{\mathrm{eff}}
  \;\xrightarrow{}\; V$
  \quad\textbf{:}\quad a self-reinforcing toxicity loop.
\end{frame}

% slide 5a: well-posedness and R_0
\begin{frame}{Well-Posedness and the Basic Reproduction Number}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Well-posedness}
        \begin{itemize}
          \item \textbf{Nonnegativity:} every face of
                $\mathbb{R}^6_{\geq 0}$ is inward-pointing
          \item \textbf{Boundedness:}
                $U \leq \nu/\lambda_u$, \;
                $C \leq \rho/\mu_c$, \;
                $\tau \leq \phi\,C_{\max}/\psi$
        \end{itemize}

        \vspace{0.3em}
        Solutions exist globally and stay in the positive cone.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Amplification threshold}
        \[
        \mathcal{R}_0
        = \frac{\alpha\,\beta_0\,I_{\mathrm{DFE}}}{\gamma_0 + \mu_c}
        \]

        \vspace{0.2em}
        \[
        \boxed{\;\alpha_{\mathrm{crit}} \approx 0.3165\;}
        \]
      \end{alertblock}

      \vspace{0.3em}
      $\mathcal{R}_0 < 1$: virality dies out (DFE stable).

      $\mathcal{R}_0 > 1$: endemic virality persists.
    \end{column}
  \end{columns}
\end{frame}

% slide 5b: bifurcation and stability
\begin{frame}{Bifurcation and Stability}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.70\textheight,keepaspectratio]{r0_bifurcation.png}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{Transcritical bifurcation}
        At $\alpha = \alpha_{\mathrm{crit}}$, the disease-free and
        endemic equilibria exchange stability.
      \end{block}

      \vspace{0.4em}
      \begin{itemize}
        \item DFE eigenvalue for $V$:
        \[
          (\gamma_0 + \mu_c)(\mathcal{R}_0 - 1)
        \]
        \item Positive equilibrium: numerically verified LAS for all
              tested $\alpha > \alpha_{\mathrm{crit}}$
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% slide 6: feedback loop
\begin{frame}{The Toxicity Feedback Loop}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\linewidth,height=0.68\textheight,keepaspectratio]{r0_bifurcation.png}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{The cycle}
        \centering
        \vspace{0.3em}
        {\large
        $V \;\xrightarrow{\;\phi\;}\; \tau
         \;\xrightarrow{\;\kappa\;}\; \beta_{\mathrm{eff}}
         \;\xrightarrow{}\; V$}
        \vspace{0.3em}
      \end{block}

      \vspace{0.4em}
      \begin{enumerate}
        \item Viral content drives discussion pressure ($\phi$)
        \item Pressure amplifies spread rate ($\kappa$)
        \item Faster spread creates more viral content
      \end{enumerate}

      \vspace{0.4em}
      \textbf{Brakes:} fatigue ($\gamma_{\mathrm{eff}}$),
      pressure decay ($\psi$), and user departure ($w\tau$).
    \end{column}
  \end{columns}
\end{frame}

% slide 10: conclusions
\begin{frame}{Conclusions}
  \begin{enumerate}
    \item \textbf{Threshold.}\;
    A sharp amplification threshold exists
    ($\alpha_{\mathrm{crit}} \approx 0.3165$).
    Below it, no sustained virality.

    \vspace{0.5em}
    \item \textbf{Feedback loop.}\;
    Above threshold, a self-reinforcing cycle locks in:
    viral content $\to$ pressure $\to$ amplified spread $\to$ more viral content.

    \vspace{0.5em}
    \item \textbf{Quantifiable tradeoff.}\;
    Aggressive amplification buys engagement today at the cost of a
    smaller, more toxic, less sustainable platform tomorrow.
  \end{enumerate}
